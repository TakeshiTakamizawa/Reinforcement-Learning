double get_reward(){
	unsigned int data[8];
	double reward;
	
	Get_IX008(0x90,data);
	
	reward=0; 
	
		
	if (data[0]>300 || data[7]>300){  
		reward=-100;
		LED(1);
	}
	else if (data[3]>300 || data[4]>300){
		reward=1; 
		LED(2);
	}
	else{
		 LED(0);
	}
	return reward;
	
}
